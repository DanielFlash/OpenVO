29 vector<Point> apply(vector<Point> points, Matrix<> H)
30 {
31 vector<Point> result;
32
33 for (int32_t i = 0; i < points.size(); i++)
34 {
35 Point point;
36 point.x = (H[0][0] * points[i].x + H[0][1] * points[i].y + H[0][2]) /
37 (H[2][0] * points[i].x + H[2][1] * points[i].y + H[2][2]);
38
39 point.y = (H[1][0] * points[i].x + H[1][1] * points[i].y + H[1][2]) /
40 (H[2][0] * points[i].x + H[2][1] * points[i].y + H[2][2]);
41
42 if (std::isinf(point.x) || std::isinf(point.y))
43 return vector<Point>();
44
45 result.push_back(point);
46 }
47
48 return result;
49 }

77pair<vector<vector<double>>, vector<double>> createMatrixEquation(vector<Point>& src,
78 vector<Point>& dst)
79{
80 size_t size = src.size();
81 vector<vector<double>> A;
82 vector<double> b;
84 for (int i = 0; i < size; i++)
85 {
86 double x = src[i].x;
87 double y = src[i].y;
88 double _x = dst[i].x;
89 double _y = dst[i].y;
90
91 vector<double> row_A1(8);
92 row_A1[0] = x;
93 row_A1[1] = y;
94 row_A1[2] = 1;
95 row_A1[3] = 0;
96 row_A1[4] = 0;
97 row_A1[5] = 0;
98 row_A1[6] = -x * _x;
99 row_A1[7] = -y * _x;
100 b.push_back(_x);
101 A.push_back(row_A1);
102
103 vector<double> row_A2(8);
104 row_A2[0] = 0;
105 row_A2[1] = 0;
106 row_A2[2] = 0;
107 row_A2[3] = x;
108 row_A2[4] = y;
109 row_A2[5] = 1;
110 row_A2[6] = -x * _y;
111 row_A2[7] = -y * _y;
112 b.push_back(_y);
113 A.push_back(row_A2);
114 }
115
116 return pair<vector<vector<double>>, vector<double>>{A, b};
117}

126vector<double> createMatrixEquationSingular(vector<Point>& src, vector<Point>& dst)
127{
128 size_t rows = 2 * src.size();
129 size_t cols = 9;
130 vector<double> A(rows * cols, 0.0);
131
132 for (int i = 0; i < src.size(); i++)
133 {
134 double x1 = src[i].x;
135 double y1 = src[i].y;
136 double x2 = dst[i].x;
137 double y2 = dst[i].y;
138
139 int i0 = 2 * i; // even line
140 A[i0 * 9 + 0] = x1;
141 A[i0 * 9 + 1] = y1;
142 A[i0 * 9 + 2] = 1.0;
143 A[i0 * 9 + 3] = 0.0;
144 A[i0 * 9 + 4] = 0.0;
145 A[i0 * 9 + 5] = 0.0;
146 A[i0 * 9 + 6] = -x2 * x1;
147 A[i0 * 9 + 7] = -x2 * y1;
148 A[i0 * 9 + 8] = -x2;
149
150 int i1 = 2 * i + 1; // odd
151 A[i1 * 9 + 0] = 0.0;
152 A[i1 * 9 + 1] = 0.0;
153 A[i1 * 9 + 2] = 0.0;
154 A[i1 * 9 + 3] = x1;
155 A[i1 * 9 + 4] = y1;
156 A[i1 * 9 + 5] = 1.0;
157 A[i1 * 9 + 6] = -y2 * x1;
158 A[i1 * 9 + 7] = -y2 * y1;
159 A[i1 * 9 + 8] = -y2;
160 }
161
162 return A;
163}

171Matrix<> calc(vector<Point>& src, vector<Point>& dst)
172{
173 pair<vector<vector<double>>, vector<double>> equation = createMatrixEquation(src, dst);
175 LuSolver solver;
176 vector<double> x = solver.solve(equation.first, equation.second);
177 // Check if the system has a solution
178 if (x.size() == 0)
179 {
180 cout << "The system of equations is degenerate and has no unique solution." << std::endl;
181 return Matrix<>(0, 0);
182 }
183
185 Matrix<> H(3, 3);
186 for (int i = 0; i < 8; i++)
187 {
188 H[i / 3][i % 3] = x[i];
189 }
190 H[2][2] = 1.0;
191
192 return H;
193}

201Matrix<> calcSingular(vector<Point> src, vector<Point> dst)
202{
203 size_t countRowEquation = src.size() * 2;
204 vector<double> equation = createMatrixEquationSingular(src, dst);
205 SvdSolver solver;
206 vector<double> x = solver.solveHomogeneous(countRowEquation, 9, equation);
207 // Checking if the system has a solution
208 if (x.size() == 0)
209 {
210 std::cout << "The system of equations is degenerate and has no unique solution.." << std::endl;
211 return Matrix<>(0, 0);
212 }
213
215 Matrix<> H(3, 3);
216 for (int i = 0; i < 9; i++)
217 {
218 H[i / 3][i % 3] = x[i] / x[8];
219 }
220
221 return H;
222}

245vector<double> calcDeltaPoint(Points src, Points dst, Matrix<> H)
246{
247 Points projection = homography_func::apply(src, H);
248 if (projection.size() == 0)
249 return vector<double>();
250
251 vector<double> vectorDelta;
252 for (size_t i = 0; i < src.size(); i++)
253 {
254 vectorDelta.push_back(pow((projection[i].x - dst[i].x), 2) +
255 pow((projection[i].y - dst[i].y), 2));
256 }
257
258 return vectorDelta;
259}

294Matrix<> findHomography(vector<Point> src, vector<Point> dst, Filters method,
295 double threshold, int maxIteration = 1000, int countRandomPoints = 4)
296{
297 if (src.size() < 4 || src.size() != dst.size())
298 {
299 throw "Count points less min need for homography";
300 return Matrix<>(0, 0);
301 }
302
303 if (threshold <= 0)
304 threshold = 0.85;
305
306 if (maxIteration <= 0)
307 maxIteration = 1000;
308
309 if (countRandomPoints < 4)
310 countRandomPoints = 4;
311
312 Matrix<> H;
313 if (method == Filters::RANSAC)
314 {
315 RANSAC ransac(threshold, maxIteration, countRandomPoints);
316 H = ransac.calc(src, dst, findMatrix, calcDeltaPoint);
317 }
318 else if (method == Filters::LMEDS)
319 {
320 LMEDS lmeds(maxIteration, countRandomPoints);
321 H = lmeds.calc(src, dst, findMatrix, calcDeltaPoint);
322 }
323
324 return H;
325}

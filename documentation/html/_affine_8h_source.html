99 vector<Point> apply(vector<Point> points, Matrix<> affine)
100 {
101 vector<Point> result;
102 for (int i = 0; i < points.size(); i++)
103 {
104 Point point;
105 point.x = affine[0][0] * points[i].x + affine[0][1] * points[i].y + affine[0][2];
106 point.y = affine[1][0] * points[i].x + affine[1][1] * points[i].y + affine[1][2];
107
108 if (std::isinf(point.x) || std::isinf(point.y))
109 return vector<Point>();
110
111 result.push_back(point);
112 }
113
114 return result;
115 }

141pair<vector<vector<double>>, vector<double>> createMatrixEquationAffine(vector<Point> src,
142 vector<Point> dst)
143{
144 size_t points = src.size();
145 size_t rows = points * 2, cols = 6;
146 vector<vector<double>> A(rows, vector<double>(cols));
147 vector<double> b(rows);
148
149 for (int i = 0; i < points; ++i)
150 {
151 double x1 = src[i].x;
152 double y1 = src[i].y;
153
154 double x2 = dst[i].x;
155 double y2 = dst[i].y;
156
157 // Equation for x'
158 int r1 = static_cast<int>(2 * i);
159 A[r1][0] = x1;
160 A[r1][1] = y1;
161 A[r1][2] = 1;
162 A[r1][3] = 0;
163 A[r1][4] = 0;
164 A[r1][5] = 0;
165 b[r1] = x2;
166
167 // Equation for y'
168 int r2 = r1 + 1;
169 A[r2][0] = 0;
170 A[r2][1] = 0;
171 A[r2][2] = 0;
172 A[r2][3] = x1;
173 A[r2][4] = y1;
174 A[r2][5] = 1;
175 b[r2] = y2;
176 }
177
178 return pair<vector<vector<double>>, vector<double>>{A, b};
179}

186Matrix<> calcAffine(pair<vector<vector<double>>, vector<double>> matrixEquation)
187{
189 LuSolver solver;
190 vector<double> x = solver.solve(matrixEquation.first, matrixEquation.second);
191 // Checking if the system has a solution
192 if (x.size() == 0)
193 {
194 std::cerr << "The system of equations is degenerate and has no unique solution.." << std::endl;
195 return Matrix<>(0, 0);
196 }
197
198 Matrix<> affineMatrix(2, 3);
199 affineMatrix[0][0] = x[0]; // a
200 affineMatrix[0][1] = x[1]; // b
201 affineMatrix[0][2] = x[2]; // tx
202 affineMatrix[1][0] = x[3]; // c
203 affineMatrix[1][1] = x[4]; // d
204 affineMatrix[1][2] = x[5]; //ty
205
206 return affineMatrix;
207}

224 affineMatrix[1][1] = x[4]; // a22

253vector<double> calcDeltaPointAffine(std::vector<Point> src, std::vector<Point> dst, Matrix<> affine)
254{
255 std::vector<Point> projection = affine_func::apply(src, affine);
256 if (projection.size() == 0)
257 return vector<double>();
258
259 vector<double> vectorDelta;
260 for (size_t i = 0; i < src.size(); i++)
261 {
262 vectorDelta.push_back(pow((projection[i].x - dst[i].x), 2) +
263 pow((projection[i].y - dst[i].y), 2));
264 }
265
266 return vectorDelta;
267}
